LOGICA ULTIMA APĂSARE (VERSIUNE ANTI-REFRESH) --- const paginaIncarcataLa = Date.now(); // Salvăm momentul exact al deschiderii paginii db.ref('jurnal').limitToLast(1).on('child_added', (snapshot) => { const acum = Date.now(); // Verificăm dacă au trecut mai mult de 3 secunde de la refresh. // Dacă DA, înseamnă că e un eveniment NOU, nu cel vechi care vine la încărcare. if (acum - paginaIncarcataLa > 3000) { const dataInregistrare = new Date(); const oraLocala = dataInregistrare.getHours().toString().padStart(2, '0') + ":" + dataInregistrare.getMinutes().toString().padStart(2, '0') + ":" + dataInregistrare.getSeconds().toString().padStart(2, '0'); const lastPressEl = document.getElementById('last-press'); lastPressEl.innerText = oraLocala; // Efect vizual de alertă la primire date noi lastPressEl.style.color = "white"; setTimeout(() => { lastPressEl.style.color = "var(--cyan)"; }, 700); } else { // Dacă e sub 3 secunde (adică e la refresh), nu scriem ora curentă. // Opțional, poți pune un text de standby: document.getElementById('last-press').innerText = "Așteptare..."; } });
